import React, { useState, useEffect, useRef } from 'react';
import * as THREE from 'three';
// We ignore the typescript error for vanta because it doesn't have official types
// @ts-ignore
import NET from 'vanta/dist/vanta.net.min';

export default function ThreeBackground({ children }: { children: React.ReactNode }) {
  const [vantaEffect, setVantaEffect] = useState<any>(null);
  const vantaRef = useRef(null);

  useEffect(() => {
    if (!vantaEffect) {
      setVantaEffect(
        NET({
          el: vantaRef.current,
          THREE: THREE,
          mouseControls: true,
          touchControls: true,
          gyroControls: false,
          minHeight: 200.00,
          minWidth: 200.00,
          scale: 1.00,
          scaleMobile: 1.00,
          color: 0xc5a059,       // Nobel Gold lines
          backgroundColor: 0x1B365D, // Mapstone Blue Background
          points: 10.00,
          maxDistance: 22.00,
          spacing: 16.00
        })
      );
    }
    return () => {
      if (vantaEffect) vantaEffect.destroy();
    };
  }, [vantaEffect]);

  return (
    <div ref={vantaRef} className="w-full h-full absolute inset-0 z-0">
      {/* The animation lives here */}
      {children}
    </div>
  );
}
